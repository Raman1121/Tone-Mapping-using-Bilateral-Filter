hdr = hdrread('...')

img = double(hdr);

% Calculate the Intensity Image
% Based on https://en.wikipedia.org/wiki/Relative_luminance
I = 0.2126 * img(:,:,1) + 0.7152 * img(:,:,2) + 0.0722 * img(:,:,3);

% Get log domain Intensity
L = log2(I);

% Get base image
B = imbilatfilt(L);

% Get Detail image
D = L - B;

% Contrast reduction only on the base
maxb = max(B(:));
minb = min(B(:)); 

% Scale base image
s = 3/(maxb-minb);
ajtB = (B-maxb).*s;
O = 2.^(ajtB+D);

% Reconstruct Color Chanel
img(:,:,1) = O.*(img(:,:,1)./I);
img(:,:,2) = O.*(img(:,:,2)./I);
img(:,:,3) = O.*(img(:,:,3)./I);

% linearly scale to 0~1
img = rescale(img);

% Apply gamma correction
img = img.^0.6;

[h,s,v] = rgb2hsv(img);
h = h.^1.2;
s = s.^1.1;
v = v.^0.7;

[rr, gg, bb] = hsv2rgb(h,s,v);
img(:,:,1) = rr;
img(:,:,2) = gg;
img(:,:,3) = bb;

end
